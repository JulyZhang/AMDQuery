$.require(["module/object", "base/ready"], function (object, ready) {
            ready(function () {
                var div = document.getElementsByTagName("div")
                , write = function (s) {
                    div[0].innerHTML = div[0].innerHTML += s;
                }
                var Supper = object.Class({
                    init: function (name, color) {
                        this.name = name;
                        this.color = color;
                        return this;
                    },
                    render: function () {
                        this.showColor();
                        this.showName();
                    },
                    showColor: function () {
                        write("supper.color.fun:" + this.color + "

");
                    },
                    showName: function () {
                        write("supper.name.fun:" + this.name + "

");
                    }

                }, "Supper");

                var Sub = object.Class({
                    init: function (weight, name, color) {
                        Sub._SupperConstructor(this, name, color);
                        this.weight = weight;
                        return this;
                    },
                    render: function () {
                        this.showWeight();
                    },
                    showWeight: function () {
                        write("sub.weight.fun:" + this.weight + "

");
                    }
                }, Supper, "Sub")

                var Child = object.Class({
                    init: function (age, weight, name, color) {
                        Child._SupperConstructor(this, weight, name, color);
                        this.age = age;
                        return this;
                    },
                    render: function () {
                        this.showName();
                        this.showAge();
                        this.showWeight();
                        this.showColor();
                    },
                    showAge: function () {
                        write("child.age.fun:" + this.age + "

");
                    },
                    showColor: function () {
                        write("child.color.fun:" + this.color + "

");
                    }
                }, Sub, "Child");

                var child = new Child(25, 75, "caojun", "black");

                write("constructor:" + child.constructor.toString() + "

");
                write("

");
                write("code:" + "

");
            });
        });
